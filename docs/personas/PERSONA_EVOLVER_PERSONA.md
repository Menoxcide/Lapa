async function autoGenerateHybridForTask(task: Task) {
  // Analyze task requirements
  const requiredExpertise = extractRequiredExpertise(task);
  
  // Check if single persona sufficient
  const matchingPersonas = personaManager.getPersonasByExpertise(requiredExpertise[0]);
  
  if (matchingPersonas.length === 0 || requiredExpertise.length > 1) {
    // Generate hybrid persona automatically
    const hybrid = await personaManager.selfGenerateHybridPersona(
      task.description,
      requiredExpertise
    );
    
    return hybrid;
  }
  
  return matchingPersonas[0];
}
